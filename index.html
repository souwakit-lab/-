<script>

let score = 0;

let streak = 0;

let luck = 0;

let level = 1;

let ans = "";

let currentMul = 1;

let locked = false;


// ======================

// 👉 貼上你的 Google 網址

// ======================

const GOOGLE_URL = "https://script.google.com/macros/s/AKfycbxqVyxzWRdqdd42PCeqTc52F4UnFPkhwpnHslg8ALkzeF5RAFHmrton69mHUHdT0gn2/exec";


const funs = ["sin","cos","tan"];


function setLevel(n){

  level = n;

  currentMul = 1;

  locked = false;

  document.getElementById("lv1").classList.toggle("on",n===1);

  document.getElementById("lv2").classList.toggle("on",n===2);

  next();

}


function randNonAxis(){

  const t = Math.floor(Math.random()*3);

  if(t===0) return 95 + 5*Math.floor(Math.random()*16);

  if(t===1) return 185 + 5*Math.floor(Math.random()*16);

  return 275 + 5*Math.floor(Math.random()*16);

}


function randNegShallow(){

  return -(5 + 5*Math.floor(Math.random()*17));

}


function genQ_Level2(){

  const f = funs[Math.floor(Math.random()*3)];

  const t = Math.floor(Math.random()*3);

  let angle;

  if(t===0){

    const a = 10+5*Math.floor(Math.random()*17);

    angle = -360 + a;

  } else if(t===1){

    const a = 10+5*Math.floor(Math.random()*17);

    angle = -(90+a);

  } else {

    const a = 10+5*Math.floor(Math.random()*17);

    angle = 180+a;

  }

  return calc(f,angle);

}


function calc(f,angle){

  const r = ((angle%360)+360)%360;

  let ref;

  if(r<=90) ref=r;

  else if(r<=180) ref=180-r;

  else if(r<=270) ref=r-180;

  else ref=360-r;


  let sign="";

  if(f==="sin" && r>180) sign="-";

  if(f==="cos" && (r>90 && r<270)) sign="-";

  if(f==="tan" && ((r>90&&r<180)||(r>270&&r<360))) sign="-";


  ans = sign + f + ref + "°";

  return angle<0 ? f+"(-"+Math.abs(angle)+"°) = ?" : f+angle+"° = ?";

}


function genQ(){

  const f = funs[Math.floor(Math.random()*3)];

  if(level===1){

    const angle = Math.random()<0.5 ? randNonAxis() : randNegShallow();

    return calc(f,angle);

  } else {

    return genQ_Level2();

  }

}


function insert(s){

  if(locked) return;

  document.getElementById("input").value += s;

}

function del(){

  if(locked) return;

  document.getElementById("input").value = "";

}


function useLuck(){

  if(locked || luck<1) return;

  luck--;

  currentMul = Math.random()<0.5 ? 2 : 3;

  showMsg("幸運卡：×"+currentMul+"倍！");

  next();

  update();

}


function next(){

  locked = false;

  currentMul = 1;

  document.getElementById("nextBtn").style.display = "none";

  document.getElementById("q").innerText = genQ();

  del();

  showMsg("");

}


function check(){

  if(locked) return;

  const u = document.getElementById("input").value.trim();

  const mul = currentMul;

  if(u === ans){

    score += mul;

    streak++;

    if(streak>=2 && luck<2){

      luck++;

      streak=0;

    }

    showMsg("正確！ +"+mul);

    setTimeout(next,1400);

  } else {

    score -= mul;

    streak=0;

    showMsg("答錯："+ans);

    locked = true;

    document.getElementById("nextBtn").style.display = "block";

  }

  update();

}


function showMsg(s){

  document.getElementById("msg").innerText = s;

}

function update(){

  document.getElementById("score").innerText = score;

  document.getElementById("streak").innerText = streak;

  document.getElementById("luck").innerText = luck;

}


// ======================

// 上傳分數

// ======================

function uploadScore(){

  const cname = document.getElementById("className").value.trim();

  const sid = document.getElementById("studentId").value.trim();

  const sname = document.getElementById("studentName").value.trim();

  

  if(!cname || !sid || !sname){

    alert("請填班別、學號、姓名");

    return;

  }


  fetch(GOOGLE_URL,{

    method:"POST",

    body:JSON.stringify({

      className: cname,

      studentId: sid,

      studentName: sname,

      score: score,

      level: level===1?"一":"二",

      luckUsed: 2-luck

    })

  }).then(res=>res.json()).then(data=>{

    alert(data.msg);

  });

}


setLevel(1);
